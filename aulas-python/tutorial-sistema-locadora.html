import os

# ================================
# FUNÇÕES BASE
# ================================

def calcular_media(avaliacoes):
    if len(avaliacoes) == 0:
        return 0
    return round(sum(avaliacoes) / len(avaliacoes), 1)


def classificar_filme(media):
    if media >= 8:
        return "🔥 Sucesso"
    elif media >= 5:
        return "👍 Regular"
    else:
        return "💀 Flop"


# ================================
# CADASTRAR FILME (ADMIN)
# ================================
def cadastrar_filme():
    titulo = input("Título do filme: ")
    genero = input("Gênero: ")

    filme = {
        "titulo": titulo,
        "genero": genero,
        "avaliacoes": [],
        "media": 0,
        "classificacao": "Sem avaliações",
        "disponivel": True,
        "cliente": None
    }

    return filme


# ================================
# EXIBIR CATÁLOGO
# ================================
def exibir_catalogo(filmes):
    print("\n===== CATÁLOGO =====\n")

    for f in filmes:
        status = "Disponível" if f["disponivel"] else f"Alugado por {f['cliente']}"

        print(f"Título: {f['titulo']}")
        print(f"Gênero: {f['genero']}")
        print(f"Média: {f['media']}")
        print(f"Avaliações: {len(f['avaliacoes'])}")
        print(f"Classificação: {f['classificacao']}")
        print(f"Status: {status}")
        print("-" * 30)


# ================================
# BUSCAR FILME
# ================================
def buscar_filme(titulo, filmes):
    for f in filmes:
        if f["titulo"].lower() == titulo.lower():
            return f
    return None


# ================================
# ALUGAR FILME
# ================================
def alugar_filme(filmes):
    titulo = input("Digite o título do filme: ")

    for f in filmes:
        if f["titulo"].lower() == titulo.lower():

            if f["disponivel"]:
                nome = input("Seu nome: ")
                f["disponivel"] = False
                f["cliente"] = nome
                print("🎬 Filme alugado!")
            else:
                print(f"❌ Já alugado por {f['cliente']}")
            return

    print("Filme não encontrado.")


# ================================
# DEVOLVER FILME + AVALIAR
# ================================
def devolver_filme(filmes):
    titulo = input("Digite o título do filme: ")

    for f in filmes:
        if f["titulo"].lower() == titulo.lower():

            if not f["disponivel"]:
                print(f"Filme estava com: {f['cliente']}")

                while True:
                    try:
                        nota = float(input("Dê uma nota (0 a 10): "))

                        if 0 <= nota <= 10:
                            break
                        else:
                            print("Nota inválida!")

                    except:
                        print("Digite um número válido!")

                f["avaliacoes"].append(nota)

                f["media"] = calcular_media(f["avaliacoes"])
                f["classificacao"] = classificar_filme(f["media"])

                f["disponivel"] = True
                f["cliente"] = None

                print("🔁 Devolvido com sucesso!")
                print(f"⭐ Nova média: {f['media']}")

            else:
                print("Esse filme já está disponível.")

            return

    print("Filme não encontrado.")


# ================================
# MELHOR E PIOR FILME
# ================================
def melhor_filme(filmes):
    return max(filmes, key=lambda f: f["media"])


def pior_filme(filmes):
    return min(filmes, key=lambda f: f["media"])


# ================================
# MENU CLIENTE
# ================================
def menu_cliente(filmes):
    while True:
        print("\n=== MENU CLIENTE 🍿 ===")
        print("1 - Ver catálogo")
        print("2 - Buscar filme")
        print("3 - Alugar filme")
        print("4 - Devolver filme")
        print("5 - Voltar")

        op = input("Escolha: ")

        if op == "1":
            if len(filmes) == 0:
                print("Nenhum filme disponível.")
            else:
                exibir_catalogo(filmes)

        elif op == "2":
            nome = input("Digite o título: ")
            filme = buscar_filme(nome, filmes)

            if filme:
                print(filme)
            else:
                print("Não encontrado.")

        elif op == "3":
            alugar_filme(filmes)

        elif op == "4":
            devolver_filme(filmes)

        elif op == "5":
            break

        else:
            print("Opção inválida!")


# ================================
# MENU ADMIN
# ================================
def menu_admin(filmes):
    senha = input("Senha do admin: ")

    if senha != "123":
        print("Acesso negado! 🔒")
        return

    while True:
        print("\n=== MENU ADMIN 🛠️ ===")
        print("1 - Cadastrar filme")
        print("2 - Ver catálogo")
        print("3 - Top e Flop")
        print("4 - Voltar")

        op = input("Escolha: ")

        if op == "1":
            filme = cadastrar_filme()
            filmes.append(filme)
            print("Filme cadastrado!")

        elif op == "2":
            exibir_catalogo(filmes)

        elif op == "3":
            if len(filmes) == 0:
                print("Sem filmes.")
            else:
                print("\n🏆 Melhor:", melhor_filme(filmes))
                print("\n💀 Pior:", pior_filme(filmes))

        elif op == "4":
            break

        else:
            print("Opção inválida!")


# ================================
# SISTEMA PRINCIPAL
# ================================
os.system("cls")

filmes = []

while True:

    print("\n=== LOCADORA 🎬 ===")
    print("1 - Cliente")
    print("2 - Admin")
    print("3 - Sair")

    op = input("Escolha: ")

    if op == "1":
        menu_cliente(filmes)

    elif op == "2":
        menu_admin(filmes)

    elif op == "3":
        print("Encerrando... 👋")
        break

    else:
        print("Opção inválida!")
