O Próximo Nível da Programação

Fala, dev! Agora que você já entende o básico de rotas, vamos dar um passo estratégico para o mercado de trabalho. Misturar comandos SQL direto dentro das rotas do site deixa o código confuso e difícil de manter.

Nesse projeto estruturado, vamos implementar o padrão DAO (Data Access Object). Além disso, utilizaremos o **Bootstrap 5** para criar uma interface limpa, responsiva e moderna de forma muito ágil!

1. Conexão Segura com Contexto (database/db.py)

Este arquivo gerencia as conexões usando o objeto especial g do Flask. O g garante que a conexão dure apenas o tempo necessário de uma requisição e se feche sozinha no final.

Código fonte de: database/db.py 
import sqlite3  # Biblioteca padrão do Python para trabalhar com SQLite
from flask import g  # Objeto do Flask para armazenar dados durante a requisição

# Nome do arquivo do banco de dados
DATABASE = "database/bdlocadora.db"

def get_db():
    # Verifica se já existe uma conexão salva no contexto da requisição
    if "db" not in g:
        # Se não existir, cria uma nova conexão com o banco
        g.db = sqlite3.connect(DATABASE)

        # Permite acessar os dados como dicionário (ex: linha["nome"])
        g.db.row_factory = sqlite3.Row

    # Retorna a conexão (nova ou já existente)
    return g.db

def close_db(e=None):
    # Remove a conexão do g (se existir) e retorna ela
    db = g.pop("db", None)

    # Se a conexão existir, fecha ela
    if db is not None:
        db.close()

def testar_conexao():
    try:
        db = get_db()  # tenta obter a conexão
        cursor = db.cursor()

        # Executa uma query simples
        cursor.execute("SELECT 1")

        print("Conexão com o banco funcionando!")
        return True
    except Exception as e:
        print("Erro ao conectar no banco:", e)
        return False

2. Isolando o SQL com o Camada DAO (database/cliente_dao.py)

O arquivo DAO é o único responsável por conversar com as colunas da nossa tabela de clientes.

Código fonte de: database/cliente_dao.py 
from database.db import get_db

def listar_todos():
    db = get_db()
    return db.execute("SELECT * FROM tbclientes ORDER BY nome").fetchall()

def salvar_novo(c):
    db = get_db()
    db.execute("""
        INSERT INTO tbclientes (nome, endereco, bairro, cidade, estado, telefone, email, senha, datanascimento)
        VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)
    """, (c['nome'], c['endereco'], c['bairro'], c['cidade'], c['estado'], c['telefone'], c['email'], c['senha'], c['datanascimento']))
    db.commit()

def buscar_por_id(id):
    db = get_db()
    return db.execute("SELECT * FROM tbclientes WHERE codcliente = ?", (id,)).fetchone()

def atualizar(id, c):
    db = get_db()
    db.execute("""
        UPDATE tbclientes 
        SET nome=?, endereco=?, bairro=?, cidade=?, estado=?, telefone=?, email=?, senha=?, datanascimento=?
        WHERE codcliente=?
    """, (c['nome'], c['endereco'], c['bairro'], c['cidade'], c['estado'], c['telefone'], c['email'], c['senha'], c['datanascimento'], id))
    db.commit()

def excluir(id):
    db = get_db()
    db.execute("DELETE FROM tbclientes WHERE codcliente = ?", (id,))
    db.commit()

3. Unindo Tudo com os Controladores de Rotas (app.py)

O arquivo app.py gerencia o envio e recebimento de dados e delega as ações de banco para o DAO. Preste atenção nas definições de rotas dinâmicas como <int:id> para evitar erros de compilação.

Código fonte de: app.py 
from flask import Flask, render_template, request, redirect, url_for
from database.db import close_db
from database import cliente_dao

app = Flask(__name__)
app.secret_key = "chave_mestra_locadora"

app.teardown_appcontext(close_db)

@app.route('/')
def home():
    return redirect(url_for('listar'))

@app.route('/clientes')
def listar():
    lista = cliente_dao.listar_todos()
    return render_template('clientes/listar-clientes.html', clientes=lista)

@app.route('/clientes/novo', methods=['GET', 'POST'])
def novo():
    if request.method == 'POST':
        dados_cliente = {
            'nome': request.form['nome'],
            'endereco': request.form['endereco'],
            'bairro': request.form['bairro'],
            'cidade': request.form['cidade'],
            'estado': request.form['estado'],
            'telefone': request.form['telefone'],
            'email': request.form['email'],
            'senha': request.form['senha'],
            'datanascimento': request.form['datanascimento']
        }
        cliente_dao.salvar_novo(dados_cliente)
        return redirect(url_for('listar'))
        
    return render_template('clientes/cadastro-clientes.html', cliente=None)

@app.route('/clientes/editar/<int:id>', methods=['GET', 'POST'])
def editar(id):
    if request.method == 'POST':
        dados_atualizados = {
            'nome': request.form['nome'],
            'endereco': request.form['endereco'],
            'bairro': request.form['bairro'],
            'cidade': request.form['cidade'],
            'estado': request.form['estado'],
            'telefone': request.form['telefone'],
            'email': request.form['email'],
            'senha': request.form['senha'],
            'datanascimento': request.form['datanascimento']
        }
        cliente_dao.atualizar(id, dados_atualizados)
        return redirect(url_for('listar'))
    
    cliente_existente = cliente_dao.buscar_por_id(id)
    return render_template('clientes/cadastro-clientes.html', cliente=cliente_existente)

@app.route('/clientes/excluir/<int:id>', methods=['POST'])
def excluir(id):
    cliente_dao.excluir(id)
    return redirect(url_for('listar'))

if __name__ == '__main__':
    app.run(debug=True)
